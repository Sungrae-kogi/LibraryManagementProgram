1. 사용자

CREATE TABLE DREAMS.MEMBER
(
    EMPNO   CHAR(7)	PRIMARY KEY,                    
    NM      VARCHAR2(25),
    PWD     VARCHAR2(100) NOT NULL,
    ROLE    VARCHAR2(10),
);

COMMENT ON COLUMN DREAMS.MEMBER.EMPNO IS '교번';
COMMENT ON COLUMN DREAMS.MEMBER.NM IS '이름';
COMMENT ON COLUMN DREAMS.MEMBER.PWD IS '비밀번호';
COMMENT ON COLUMN DREAMS.MEMBER.ROLE IS '권한';

2. 도서

CREATE TABLE DREAMS.BOOKS
(
    BOOK_ID CHAR(8),
    DUPL    NUMBER(2),    
    TITLE   VARCHAR2(250),
    ISBN    VARCHAR2(250),
    AUTHOR  VARCHAR2(100),
    IN_DT   DATE
);

COMMENT ON COLUMN DREAMS.BOOKS.BOOK_ID IS '도서 번호';
COMMENT ON COLUMN DREAMS.BOOKS.DUPL IS '복본 번호';
COMMENT ON COLUMN DREAMS.BOOKS.TITLE IS '책 제목';
COMMENT ON COLUMN DREAMS.BOOKS.ISBN IS 'isbn';
COMMENT ON COLUMN DREAMS.BOOKS.AUTHOR IS '저자';
COMMENT ON COLUMN DREAMS.BOOKS.IN_DT IS '수서일';


3. 대출반납

CREATE TABLE DREAMS.RENT
(
    RENT_ID CHAR(8),
    BOOK_ID CHAR(8),
    EMPNO   CHAR(7),
    BOR_DT  DATE,
    RET_DT  DATE
);

COMMENT ON COLUMN DREAMS.RENT.RENT_ID IS '대출반납 번호';
COMMENT ON COLUMN DREAMS.RENT.BOOK_ID IS '도서 번호';
COMMENT ON COLUMN DREAMS.RENT.EMPNO IS '교번';
COMMENT ON COLUMN DREAMS.RENT.BOR_DT IS '대출일';
COMMENT ON COLUMN DREAMS.RENT.RET_DT IS '반납일';
